\noindent{\bf Задача.}
{\it
Найдите длину вектора $\vec{a}\{2;3\}$
}\\
\noindent{\bf Дано:}\\
$\vec{a}\{2;3\}$
\\
\noindent{\bf Найти:}
$|\vec{a}|$.
\\
\begin{solution}
$|\vec{a}|=\sqrt{x^2 + y^2}=\sqrt{2^2 + 3^2}=\sqrt{4+9}=\sqrt{13}=3,605551$
\end{solution}
